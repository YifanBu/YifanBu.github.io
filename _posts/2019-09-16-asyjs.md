# Callbacks

```javascript
const posts = [
  { title: 'Post One', body: 'This is post one' },
  { title: 'Post Two', body: 'This is post two' }
];

function getPosts() {
  setTimeout(() => {
    let output = '';
    posts.forEach(() => {
      output += `<li>${post.title}<li>`;
    });
    document.body.innerHTML = output;
}, 1000);
}

function createPost(post, callback) {
  setTimeout(() => {
    posts.push(post);
    callback();
  }, 2000);
}

createPost({ title:'Post Three', body: 'This is post three'}, getPosts);
```

# Promises

## Promise

```javascript
const posts = [
  { title: 'Post One', body: 'This is post one' },
  { title: 'Post Two', body: 'This is post two' }
];

function getPosts() {
  setTimeout(() => {
    let output = '';
    posts.forEach(() => {
      output += `<li>${post.title}<li>`;
    });
    document.body.innerHTML = output;
}, 1000);
}

function createPost(post) {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      posts.push(post);

      const error = false;

      if(!error) {
        resolve();
      } else {
        reject('Error: Something went wrong');
      }
    }, 2000);
  });  
}

createPost({ title:'Post Three', body: 'This is post three' })
  .then(getPosts)
  .catch(err => console.log(err));
```

## Promise.all

*Promise.all()* takes an array of promises.

```javascript
const promise1 = Promise.resolve('Hello World');
const promise2 = 10;
const promise3 = new Promise((resolve, reject) =>
  setTimeout(resolve, 2000, 'Goodbye')
);
const promise4 = fetch('').then(res => res.json());

Promise.all([promise1, promise2, promise3, promise4]).then(values => console.log(values);)
```

# AJAX

## What is AJAX

- Asynchronous JavaScript & XML(Extensive Markup Language)
- Set of web technologies
- Send & receive data(request & response) asynchronously
- Does not interfere with current web page
- JSON has replaced XML for the most part

## Why use AJAX

AJAX allows us to make requests asynchronously in the background without having to reload or refresh the entire web page, which is **much faster and more interactive than reloading the entire page**.

## Libraries & Other Methods

- jQuery: a DOM Manipulation Library
- Axios: really nice to make http calls
- Superagent: really nice to make http calls too
- Fetch API: advanced browser feature
- Prototype
- Node HTTP

## Get Text File

```html
<button id="button">Get Text File</button>

<script>
  
  document.getElementById('button').addEventListener('click', loadText);
  
  function loadText(){
    // Create XHR(XMLHttpRequest) Object
    var xhr = new XMLHttpRequest();
    // OPEN - type of request, url/filename, async
    xhr.open('GET', 'sample.text', true);

    xhr.onload = function(){
      // 200: 'OK', 403: 'Forbidden', 404: 'Not Found'
      if(this.status == 200){
        document.getElementById('text').innerHTML = this.responseText;
      } elseif(this.state = 404){
        document.getElementById('text').innerHTML = 'Not Found';
      }
    }

    // another method
    xhr.onreadystatechange = function(){
      if(this.readyState == 4 && this.status == 200){
        console.log(this.responseText);
      }
    }

    // Send request
    xhr.send();
  }
</script>
```

## Get Local JSON

```html
<button id="button">Get User</button>
<br>
<h1>User</h1>
<div id="user"></div>

<script>
  
  document.getElementById('button').addEventListener('click', loadUser);

  function loadUser(){
    var xhr = new XMLHttpRequest();
    xhr.open('GET', 'user.json', true);

    xhr.onload = function(){
      if(this.status == 200){
        var user = JSON.parse(this.responseText);

        document.getElementById('user').innerHTML = user;
      }
    }

    xhr.send();
  }
</script>
```

# Fetch API

Fetch data from an external API called *JSONPlaceholder API*.

```html
<button id="getText">Get Text</button>
<div id="output"></div>

<script>
  
  document.getElementById('getText').addEventListener('click', getText);

  function getText(){
    fetch('sample.txt')
    .then((res) => res.text())
    .then((data) => {
      document.getElementById('output').innerHTML = data;
    })
    .catch((err) => console.log(err))
  }
</script>
```
